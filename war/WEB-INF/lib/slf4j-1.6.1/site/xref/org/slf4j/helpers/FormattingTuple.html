22  package org.slf4j.helpers;
23  
24  /**
25   * Holds the results of formatting done by {@link MessageFormatter}.
26   * 
27   * @author Joern Huxhorn
28   */
29  public class FormattingTuple {
30    
31    
32    static public FormattingTuple NULL = new FormattingTuple(null);
33    
34    private String message;
35    private Throwable throwable;
36    private Object[] argArray;
37    
38    public FormattingTuple(String message) {
39      this(message, null, null);
40    }
41  
42    public FormattingTuple(String message, Object[] argArray, Throwable throwable) {
43      this.message = message;
44      this.throwable = throwable;
45      if(throwable == null) {
46        this.argArray = argArray;
47      } else {
48        this.argArray = trimmedCopy(argArray);
49      }
50    }
51  
52    static Object[] trimmedCopy(Object[] argArray) {
53      if(argArray == null || argArray.length == 0) {
54        throw new  IllegalStateException("non-sensical empty or null argument array");
55      }
56      final int trimemdLen = argArray.length -1;
57      Object[] trimmed = new Object[trimemdLen];
58      System.arraycopy(argArray, 0, trimmed, 0, trimemdLen);
59      return trimmed;
60    }
61    
62    public String getMessage() {
63      return message;
64    }
65  
66    public Object[] getArgArray() {
67      return argArray;
68    }
69  
70    public Throwable getThrowable() {
71      return throwable;
72    }
73  
74  }
